CREATE TABLE usuarios (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
    senha VARCHAR(255) NOT NULL,
    telefone VARCHAR(20),
    sexo ENUM('masculino', 'feminino', 'outro'),
    data_nasc DATE,
    cidade VARCHAR(50),
    estado VARCHAR(50),
    endereco VARCHAR(150)
);
